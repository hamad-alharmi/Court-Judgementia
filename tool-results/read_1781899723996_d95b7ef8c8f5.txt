   282→   282→      const data = await res.json();

   284→   284→      setGeneratedScenario(sc);
   285→   285→      // persist scenarioId so the other client fetches the same scenario
   286→   286→      await update({ scenarioId: sc.id });

   288→   288→    } catch (e) {
   289→   289→      console.error("case gen failed", e);

   291→   291→      return null;
   292→   292→    } finally {
   293→   293→      setGenLoading(false);

   370→   370→  async function runAITurn() {
   371→   371→    if (!room || !scenario) return;
   372→   372→    const key = `ai-${room.phase}-${room.gameState.currentRound}`;
   373→   373→    if (doneRef.current.has(key)) return;
   374→   374→    doneRef.current.add(key);
   375→   375→    const side = room.phase === "prosecutor_turn" ? "prosecutor" : "defense";
   376→   376→    const argText = generateAIArgument(
   377→   377→      side,
   378→   378→      scenario,
   379→   379→      scenario.evidence,
   380→   380→      room.matchmakingType === "ranked",
   381→   381→      room.aiDifficulty,
   382→   382→      room.gameState.currentRound,
   383→   383→      room.statementCount,
   384→   384→      room.gameState.statements,
   385→   385→    );
   386→   386→    const favored = scenario.evidence.filter((e) =>
   387→   387→      side === "prosecution" ? e.side !== "defense" : e.side !== "prosecution",
   388→   388→    );
   389→   389→    const offset = (room.gameState.currentRound - 1) % Math.max(1, favored.length);
   390→   390→    const aiEvs = [favored[offset]?.id, favored[(offset + 1) % Math.max(1, favored.length)]?.id].filter(
   391→   391→      Boolean,
   392→   392→    ) as string[];
   393→   393→    const stmt: Statement = {
   394→   394→      id: "stmt-ai-" + Math.random().toString(36).slice(2, 10),
   395→   395→      round: room.gameState.currentRound,
   396→   396→      side,
   397→   397→      text: argText,
   398→   398→      evidenceIds: aiEvs,
   399→   399→      objections: [],
   400→   400→      at: Date.now(),
   401→   401→      authorIsAI: true,
   402→   402→    };
   403→   403→    const nextStatements = [...room.gameState.statements, stmt];
   404→   404→    const nextPhase = nextPhaseAfterStatement(
   405→   405→      nextStatements.map((s) => ({ side: s.side, round: s.round })),
   406→   406→      room.statementCount,
   407→   407→    );
   408→   408→    const nextRound =
   409→   409→      nextPhase === "prosecutor_turn" && side === "defense"
   410→   410→        ? room.gameState.currentRound + 1
   411→   411→        : room.gameState.currentRound;
   412→   412→    await update({
   413→   413→      gameState: {
   414→   414→        ...room.gameState,
   415→   415→        statements: nextStatements,
   416→   416→        currentRound: nextRound,
   417→   417→        turnStartedAt: nextPhase === "jury_voting" ? null : Date.now(),
   418→   418→        turnTimerRemaining: nextPhase === "jury_voting" ? 0 : TURN_DURATION,
   419→   419→      },
   420→   420→      phase: nextPhase,
   421→   421→    });
   422→   422→    toast.info(`${side === "prosecution" ? "Prosecution" : "Defense"} AI filed (R${stmt.round}).`);
   423→   423→  }

   425→   425→  async function runObjection(
   426→   426→    objectorSide: "prosecutor" | "defense",
   427→   427→    objectorName: string,
   428→   428→    targetIndex: number,
   429→   429→    grounds: string,
   430→   430→    isAI: boolean,
   431→   431→  ) {
   432→   432→    if (!room || !scenario) return;
   433→   433→    const left = room.gameState.objectionsLeft[objectorSide];
   434→   434→    if (left <= 0) {
   435→   435→      toast.error("No objections remaining.");
   436→   436→      return;
   437→   437→    }
   438→   438→    // call judge
   439→   439→    setJudging(true);
   440→   440→    try {
   441→   441→      const res = await fetch("/api/judge", {
   442→   442→        method: "POST",
   443→   443→        headers: { "Content-Type": "application/json" },
   444→   444→        body: JSON.stringify({
   445→   445→          action: "objection",
   446→   446→          scenario,
